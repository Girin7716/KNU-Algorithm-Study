- 말의 위치상황(말이 4마리이므로 원소도 4개)

이렇게 초기 구성이 끝나면 재귀를 통해서 값을 찾아가면 된다.

```python
def go(horse, count, result, state):
    global gridDict, scoreDict, answer, inputArr
    copyState = state.copy()        # 말 위치
    nowDice = inputArr[count] - 1   # 현재 주사위 숫자
    copyState[horse] = gridDict[copyState[horse]][nowDice]  # 현재 말의 다음 위치는 현재 말의 위치에서의 주사위 던진 곳
    nowScore = scoreDict[copyState[horse]]  # 점수는 다음 위치할 말의 점수

    if count == 9: answer = max(answer, result + nowScore); return  # 주사위를 다 던졌다면 기존의 값과 움직인 것을 더한 값을 비교해서 업데이트

    nextDice = inputArr[count + 1] - 1  # 다음 주사위 값 가져오기
    for i in range(4):  # 말들 전부 조사
        if copyState[i] == -1: continue # 해당 말이 도착했으면 그만
        # 현재 말의 다음 장소가 도착이 아니고 현재말의 다음장소가 말들이 존재하면 그만
        if gridDict[copyState[i]][nextDice] != -1 and gridDict[copyState[i]][nextDice] in copyState: continue

        go(i, count + 1, result + nowScore, copyState)  # 현재 말, 다음 주사위, 현재 점수, 말들의 위치
```
